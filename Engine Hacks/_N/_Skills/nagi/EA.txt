ORG 0x2a2bc //setFlag
BYTE	$00 $49 $8F $46
POIN	nagi_setFlag

	jumpToHack(nagi_kaze)

//追撃なし
ORG 0x02af60
  jumpToHack(nagi_tome)

	ORG 0x02b866
	SHORT $46C0
	SHORT $2000

afterAttack:

	ALIGN 4

	ORG 0x01858c
	jumpToHack(afterBattle)
POP
afterBattle:
	#incbin afterBattle.dmp
	POIN hasQuickenedPulse

	ORG 0x01e684
	jumpToHack(grafix_ws)
POP
grafix_ws:
	#incbin grafix.dmp
ALIGN 4
